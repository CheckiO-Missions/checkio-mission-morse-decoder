Отже, згідно з умовою, наш результат повинен складатися зі слів, а не з букв. Нам потрібно використовувати ''.join() тут є приклад.

result = []
for cod in code.split('   '):
    new = []
    for i in cod.split():
        new.append(MORSE[i])

                                 # Після цього ми повертаємося до третього рядка, де наш

def morse_decoder(code: str) -> str:
    result = []
    for cod in code.split('   '):  # В змінній cod ми передаємо зашифроване слово
        new = []  # Сюди ми помістимо вже розшифроване слово. Ця змінна скинеться
                  # коли завершиться цикл нижче
        for i in cod.split():
            new.append(MORSE[i])  # Тут ми кожну літеру відшукуємо
                                  # в азбуці Морзе і повертаємо у new
        result.append(''.join(new))  # Додаємо слово до нашого result у вигляді рядка
    return ' '.join(result).capitalize()  # Повертаємо результат у вигляді рядка з
                                          # великої букви

Або ти можеш використовувати однорядкову форму:

def morse_decoder(code: str) -> str:
    return ' '.join(''.join([MORSE[i] for i in cod.split()])
                    for cod in code.split('   ')).capitalize()
