def morse_decoder(code: str) -> str:
    result = []
    for cod in code.split('   '):  # In the cod variable we pass the encrypted word
        new = []  # Here we will put the already decoded word. This variable will be
                  # reset when the loop below completes
        for i in cod.split():
            new.append(MORSE[i])  # Here we find each letter in the Morse code and
                                  # return it to new
        result.append(''.join(new))  # Add a word to our result as a string
    return ' '.join(result).capitalize()  # We return the result as a string with an
                                          # uppercase letter
